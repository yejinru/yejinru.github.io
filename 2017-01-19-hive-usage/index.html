3. UDF

  1. Python

    --transform必须独占一次select操作
set hivevar:udf_file=cate_merge.py;
set hivevar:udf_path=/home/yejinru/${udf_file};
add file ${udf_path};
select
    transform(user_id) using 'python ${udf_file}' as (user_id string, user_type BIGINT)
from table_test;

  2. Java

    add jar /home/yejinru/brickhouse-0.7.1-SNAPSHOT.jar;
create temporary function collect AS 'brickhouse.udf.collect.CollectUDAF';

select
    user_id,
    collect(score) as score_vector
from table_test;

4. 读写文件

1. 结果输出到文件

    insert overwrite local directory '/home/yejinru/output.xsl'
    row format delimited 
    fields terminated by '\t'
    lines terminated by '\n'
    select
        concat_ws('=', 'uid', t.user_id)
    from (
        select
            user_id
        from user
        where log_date = '2017-01-19'
    ) t;

2. 从本地文件转载到hive表

    load data local inpath '/home/yejinru/input.log'
    into table table_test partition(log_date='2017-01-19');
